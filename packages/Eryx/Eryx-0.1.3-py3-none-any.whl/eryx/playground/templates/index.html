      Example

      const timenow = time();

func add (x, y) {
    let result = x + y;

    func sub(a, b) {
        return a - b;
    }

    let result2 = sub(y, x);

    let obj2 = {
        x,
        y
    };

    let five;
    five = 10 - 5;
    const neg_num = -1000;
    let flt = 69.420;
    const string = "Strings work!!!!";

    let obj = {
        five,
        neg_num,
        flt,
        result2,
        timenow,
        result,
        object: obj2,
        string,
        arr: [1, 2, 3, 4, 5]
    };

    return obj;
}

print(add(5, 7))

func makeAdder(offset) {
    func add(x, y) {
        return x + y + offset;
    }

    return add;
}

let adder = makeAdder(10);
print(adder(10, 5))
